<body>
  <h1>👁️ MILADY DEAD EYE STARE</h1>
  <p>Stare at the negative for 30 seconds, then look at a white wall</p>

  <input type="file" id="fileInput" accept="image/*" style="display:none">
  
  <button onclick="document.getElementById('fileInput').click()">Load Image</button>
  <button onclick="generateNegative()">Generate Negative</button>

  <div>
    <h3>Original</h3>
    <img id="original" alt="Original">
  </div>

  <div>
    <h3>Dead Eye Stare</h3>
    <img id="negative" alt="Negative Image">
    <div id="timer"></div>
  </div>

  <script>
    let currentImage = null;
    let timerInterval = null;

    document.getElementById('fileInput').addEventListener('change', function(e) {
      const file = e.target.files[0];
      if (!file) return;

      const reader = new FileReader();
      reader.onload = function(event) {
        currentImage = event.target.result;
        document.getElementById('original').src = currentImage;
      };
      reader.readAsDataURL(file);
    });

    function generateNegative() {
      if (!currentImage) {
        alert("Please load an image first!");
        return;
      }

      const img = new Image();
      img.src = currentImage;
      
      img.onload = function() {
        const canvas = document.createElement('canvas');
        canvas.width = img.width;
        canvas.height = img.height;
        const ctx = canvas.getContext('2d');
        
        ctx.drawImage(img, 0, 0);
        const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
        const data = imageData.data;

        // ✅ Correct color inversion
        for (let i = 0; i < data.length; i += 4) {
          data[i]     = 255 - data[i];     // Red
          data[i + 1] = 255 - data[i + 1]; // Green
          data[i + 2] = 255 - data[i + 2]; // Blue
        }

        ctx.putImageData(imageData, 0, 0);
        document.getElementById('negative').src = canvas.toDataURL();

        startTimer();
      };
    }

    function startTimer() {
      if (timerInterval) clearInterval(timerInterval);
      
      let timeLeft = 30;
      const timerElement = document.getElementById('timer');
      
      timerElement.textContent = timeLeft;
      
      timerInterval = setInterval(() => {
        timeLeft--;
        timerElement.textContent = timeLeft;
        
        if (timeLeft <= 0) {
          clearInterval(timerInterval);
          timerElement.textContent = "Look at a white wall now 👀";
        }
      }, 1000);
    }
  </script>
